// Java-Version des Rechners
import java.util.Scanner;

public class Taschenrechner{
    public static void main(String[] args){
        Scanner scanner = new Scanner(System.in);
        System.out.println("Welche Funktion willst du nutzen \n Normal = normales rechnen \n Ges = Geschwindigkeit\n BMI ");
        String modus = scanner.next();
        System.out.println(modus.toUpperCase());
        if(modus.toUpperCase().equals("GES")){
            System.out.println("Wie lang ist die Distanz");
            double strecke = NutzerEingabe(scanner,"Die strecke kann nicht null sein");
            System.out.println("Wie viel zeit hat er gebraucht");
            double zeit = NutzerEingabe(scanner,"Die Zeit kann nicht null sein");
            String ges = BerechneGes(scanner,strecke,zeit);
            System.out.println(ges);
        }    
        else if(modus.toUpperCase().equals("BMI")){
            System.out.println("Wie hoch ist dein Gewicht");
            double gewicht = NutzerEingabe(scanner,"Dein Gewicht kann nicht null sein");
            System.out.println("Wie groß bist du");
            double groese = NutzerEingabe(scanner,"Deine größe kann nicht null sein");
            String Bmi = BerechneBmi(scanner,gewicht,groese);
            System.out.println(Bmi);
        }
        else{
            System.out.println("Als Default wird immer Normal gewählt gib deine erste zahl ein");
            double ergebnis = HoleNutzerEingabe(scanner,' ');
            while(true){
                System.out.println("Gib mir ein Operator (+,-,*,/,=) ");
                String eingabe = scanner.next();
                char operator=eingabe.charAt(0); 
                if(eingabe.equals("=")){
                    break;
                }
                if(!IstValiderOperator(operator)){
                    System.out.println("Gib einen der validen Operatoren an");
                    continue;
                }
                double zahl = HoleNutzerEingabe(scanner,operator);

                ergebnis = BerechneErgebnisAusEingabe(ergebnis, zahl, operator); 
            }
            System.out.println("Ergebnis:  "+ ergebnis);
        }
    }
    private static double HoleNutzerEingabe(Scanner scanner, char operator){
        double zahl=0;
        boolean richtigeEingabe= false;
        while(!richtigeEingabe){
            try{
                System.out.println("Gib mir eine Zahl bitte");
                zahl=scanner.nextDouble();
                if(operator=='/' && zahl==0){
                    System.out.println("Durch 0 teilen nicht möglich");
                }
                else{
                    richtigeEingabe= true;
                }
            }
            catch(Exception e){
                System.out.println("Fehler: "+e);
                scanner.next();
            }
        }   
        return zahl;
    }
    private static double NutzerEingabe(Scanner scanner,String text){
        double zahl=0;
        boolean richtigeEingabe= false;
        while(!richtigeEingabe){
            try{
                System.out.println("Gib mir eine Zahl bitte");
                zahl=scanner.nextDouble();
                if(zahl==0){
                    System.out.println(text);
                }
                else{
                    richtigeEingabe= true;
                }
            }
            catch(Exception e){
                System.out.println("Fehler: "+e);
                scanner.next();
            }
        }   
        return zahl;
    }
    private static boolean IstValiderOperator(char operator){
        return operator == '+' ||
        operator=='-'||
        operator=='*'||
        operator=='/';
    }
 
    private static double BerechneErgebnisAusEingabe(double zahl1, double zahl2, char operator){
        double ergebnis = 0;
        if(operator=='+'){
            ergebnis = zahl1+zahl2;
        }
        else if(operator=='-'){
            ergebnis = zahl1-zahl2;
        }
        else if(operator=='*'){
            ergebnis = zahl1*zahl2;
        }
        else if(operator=='/'){
            ergebnis = zahl1/zahl2;
        }
        return ergebnis ;
    }
    private static String BerechneGes(Scanner scanner ,double strecke , double zeit){
            System.out.println("Für Km/h eine 1 und für m/s eine 2");
            String eingabe = scanner.next();
            double ges = strecke/zeit;
            if(eingabe.equals("1")){
                System.out.println("Willst du auch m/s haben");
                eingabe = scanner.next();
                    if (eingabe.toUpperCase().equals("JA")){
                        return ges+"Km/H  "+ges*3.6+"M/s";
                }
                    else{
                        return ges+"Km/H";
                }
            }
            else{
                System.out.println("Willst du auch Km/H haben");
                eingabe = scanner.next();
                    if (eingabe.toUpperCase().equals("JA")){
                        return ges+"m/s  "+ges/3.6+"Km/h";
                }
                    else{
                        return ges+"m/s";
                }
            }
    }
    private static String BerechneBmi(Scanner scanner ,double gewicht , double groese){
            System.out.println("Bist du als Männliches(M) oder weibliches(W) wesen geboren");
            String geschlecht = scanner.next();
            double Bmi = gewicht / (groese*groese);
            if(geschlecht.toUpperCase().equals("M")){
                if (Bmi < 20) {
                    return "Du hast Untergewicht dein BMI ist"+Bmi;
                }else if (Bmi < 25) {
                    return "Du hast Normalgewicht dein BMI ist"+Bmi;
                }else if (Bmi < 30) {
                    return "Du hast Übergewicht dein BMI ist"+Bmi;
                }else {
                    return "Du hast Adipositas dein BMI ist"+Bmi;
                }
            }
            if(geschlecht.toUpperCase().equals("W")){
                if (Bmi < 19) {
                    return "Du hast Untergewicht dein BMI ist"+Bmi;
                }else if (Bmi < 24) {
                    return "Du hast Normalgewicht dein BMI ist"+Bmi;
                }else if (Bmi < 30) {
                    return "Du hast Übergewicht dein BMI ist"+Bmi;
                }else {
                    return "Du hast Adipositas dein BMI ist"+Bmi;
                }
            }
            else{
                return "Du hast ein falsches geschlecht eingegeben hier ist dein BMI:"+Bmi;
            }
            }
  }
